    # 六种继承方式

    # 1.原型继承

    核心：子类通过增加原型链的连接继承父类的所有属性和方法（公+私）作为子类的公有属性和方法

    function Animal() {
  this.name = 'cat'
}
Animal.prototype.getName = function() {
  console.log(this.name)
}

function Cat() {
  this.name = 'lihua'
}
Cat.prototype = new Animal(); // 原型继承
Cat.prototype.constructor = Cat;

var cat = new Cat();
cat.getName() // lihua

    # 2.构造函数继承

    核心：将父类的私有属性和方法克隆一份给子类，作为子类的私有属性，与父类没有关系

    function Animal() {
  this.name = 'cat'
}
Animal.prototype.getName = function() {
  console.log(this.name)
}

function Cat() {
  Animal.call(this) // 普通函数的call调用
}

var cat = new Cat();
cat.name; // cat
cat.getName() // typeError: cat.getName is not a function

    # 3.冒充对象继承

    核心：将父类的属性和方法（私有+公有）克隆一份给子类，作为子类的私有属性，与父类没有关系

    function Animal() {
  this.name = 'cat'
}
Animal.prototype.getName = function() {
  console.log(this.name)
}

function Cat() {
  var temp = new Animal();
  for (var k in temp) {
    this[k] = temp[k]
  }
  temp = null;
}

var cat = new Cat();
cat.name; // cat
cat.getName()

    # 4.寄生组合继承

    核心：将父类的属性和方法（私有+公有）克隆一份给子类，作为子类的私有属性，与父类没有关系 缺点：父类构造函数被调用链两次

    function Animal() {
  this.name = 'cat'
}
Animal.prototype.getName = function() {
  console.log(this.name)
}

function Cat() {
  Animal.call(this); // 构造函数继承父类私有属性
}
Cat.prototype = new Animal(); // 原型链继承父类公私有属性为自己公有属性
Cat.prototype.constructor = Cat;

var cat = new Cat();
cat.name; // cat
cat.getName()

    # 5.寄生组合继承优化

    核心：将父类的属性和方法（私有+公有）克隆一份给子类，作为子类的私有属性，与父类没有关系 在寄生组合的基础上减少构造函数的调用

    function Animal() {
  this.name = 'cat'
}
Animal.prototype.getName = function() {
  console.log(this.name)
}

function Cat() {
  Animal.call(this); // 构造函数继承父类私有属性
}
Cat.prototype = Object.create(Animal.prototype); // 原型链继承父类公私有属性为自己公有属性
Cat.prototype.constructor = Cat;

var cat = new Cat();
cat.name; // cat
cat.getName()
